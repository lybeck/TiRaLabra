\usepackage{algorithm}
\usepackage{algpseudocode}


\begin{document}


\begin{algorithm}
\caption{The addition function}
\begin{algorithmic}[1]
\Function {Add}{$M_1$, $M_2$}

\State $m = M_1.rows$
\State $n = M_1.cols$

\State $A$ = new $Matrix(m,n)$

\For{$i = 1$ to $m$}
	\For{$j = 1$ to $n$}
		\State $A(i,j)$ = $M_1(i,j) + M_2(i,j)$
	\EndFor
\EndFor

\State \Return $A$

\EndFunction
\end{algorithmic}
\end{algorithm}


\begin{algorithm}
\caption{The naive multiplication function}
\begin{algorithmic}[1]
\Function {Mul}{$M_1$, $M_2$}

\State $m = M_1.rows$
\State $n = M_1.cols$
\State $p = M_2.cols$

\State $A$ = new $Matrix(m,p)$

\For{$i = 1$ to $m$}
	\For{$j = 1$ to $p$}
		\State $A(i,j) = 0$
		\For{$k = 1$ to $n$}
			\State $A(i,j) = M_1(i,k) \cdot M_2(k,j)$
		\EndFor
	\EndFor
\EndFor

\State \Return $A$

\EndFunction
\end{algorithmic}
\end{algorithm}
